<div class="grid-container">
  <div class="column-10 pre-1 post-1 padding-leader-6">
    <a class="image-switcher js-view-toggle is-active"
       data-set="image-switcher"
       data-view="foo"
       href="#">
      <figure class="image-switcher-preview">
        <img src="/assets/img/illustrations/01-circle.svg" alt="">
      </figure>
      <div class="image-switcher-title">
        <p class="trailer-0">Image One</p>
      </div>
    </a>

    <a class="image-switcher js-view-toggle"
       data-set="image-switcher"
       data-view="bar"

      </figure>

        <p class="trailer-0">Image Two</p>
      </div>
    </a>

    <a class="image-switcher js-view-toggle"
       data-set="image-switcher"
       data-view="baz"
       href="#">

Carousel

<div class="carousel">
  <div class="carousel-panel is-active" data-toggle-set="carousel" data-toggle="foo">
    <img src="/assets/img/illustrations/01-picture.svg" alt="one">
  </div>
  <div class="carousel-panel" data-toggle-set="carousel" data-toggle="baz">
    <img src="/assets/img/illustrations/02-picture.svg" alt="two">
  </div>
  <div class="carousel-panel" data-toggle-set="carousel" data-toggle="bar">
    <img src="/assets/img/illustrations/03-picture.svg" alt="three">
  </div>
</div>

<nav class="carousel-nav">

     data-toggle-set="carousel"
